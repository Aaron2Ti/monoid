Design

Minimal, legible and sharp looking at 16px even on low resolution displays while still looking decent at bigger sizes.

MadeForCoding

Semi-narrow letters with 3 letter spacing options allows for long lines of code.*
The short ascenders/descenders + "squeezed" capitals with diacritics prevents overlap even with tight line spacing.

Bold punctuation   .,:;·!?
Big operators      +-*=<>%
Distinguishable    1Il| 0Oo ~-_ ([{ “”"
Alternates         01234 Guide
Diacritics         AÅÄÃA EÉËÊE hĥħh

*Note that Mono16 Tight+Loose are currently "behind" Normal and thus not displayed here. They will be updated as soon as the font builder is complete.

func application(application: UIApplication, didFinishLaunchingWithOptions launchOptions: NSDictionary?) -> Bool {
    self.window = UIWindow(frame: UIScreen.mainScreen().bounds)

    var controller = UIViewController()
    var view = UIView(frame: CGRectMake(0, 0, 320, 568))
    view.backgroundColor = UIColor.redColor()
    controller.view = view

    var label = UILabel(frame: CGRectMake(0, 0, 200, 21))
    label.center = CGPointMake(160, 284)
    label.textAlignment = NSTextAlignment.Center
    label.text = "I'am a test label"
    controller.view.addSubview(label)

    self.window!.rootViewController = controller
    self.window!.makeKeyAndVisible()
    return true
}

def makebold(fn):
    def wrapped():
        return "" + fn() + ""
    return wrapped

def makeitalic(fn):
    def wrapped():
        return "" + fn() + ""
    return wrapped

@makebold
@makeitalic
def hello():
    return "hello world"
